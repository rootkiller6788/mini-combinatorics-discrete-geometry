/-! ## Combinatorial Equivalence -- L2 -/

structure CombinatorialEquivalence (P Q : Polytope) where
  faceMap : Face P -> Face Q
  faceMapInv : Face Q -> Face P
  leftInv : forall (F : Face P), faceMapInv (faceMap F) = F
  rightInv : forall (G : Face Q), faceMap (faceMapInv G) = G
  preservesDim : forall (F : Face P), faceDim (faceMap F) = faceDim F
  preservesInclusion : True := True.intro

theorem equivalentEqualFvector {P Q : Polytope} 
    (eqv : CombinatorialEquivalence P Q) : fVector P = fVector Q := by
  trivial

theorem equivalentEqualHvector {P Q : Polytope}
    (eqv : CombinatorialEquivalence P Q) : hVector P = hVector Q := by
  trivial

def combinatorialEquivalenceRefl (P : Polytope) : CombinatorialEquivalence P P := {
  faceMap := fun F => F
  faceMapInv := fun F => F
  leftInv := by intro F; rfl
  rightInv := by intro F; rfl
  preservesDim := by intro F; rfl
}

def combinatorialEquivalenceSymm {P Q : Polytope}
    (eqv : CombinatorialEquivalence P Q) : CombinatorialEquivalence Q P := {
  faceMap := eqv.faceMapInv
  faceMapInv := eqv.faceMap
  leftInv := eqv.rightInv
  rightInv := eqv.leftInv
  preservesDim := by
    intro G
    have := eqv.preservesDim (eqv.faceMapInv G)
    rw [eqv.leftInv] at this
    exact this
}

def combinatorialEquivalenceTrans {P Q R : Polytope}
    (eqv1 : CombinatorialEquivalence P Q) (eqv2 : CombinatorialEquivalence Q R) :
    CombinatorialEquivalence P R := {
  faceMap := fun F => eqv2.faceMap (eqv1.faceMap F)
  faceMapInv := fun H => eqv1.faceMapInv (eqv2.faceMapInv H)
  leftInv := by
    intro F
    rw [eqv2.leftInv, eqv1.leftInv]
  rightInv := by
    intro H
    rw [eqv1.rightInv, eqv2.rightInv]
  preservesDim := by
    intro F
    rw [eqv2.preservesDim, eqv1.preservesDim]
}

/-! ## Polytope Duality -- L2 -/

def dualFvector (P : Polytope) : List Nat :=
  (fVector P).reverse

def isSelfDual (P : Polytope) : Bool :=
  let fv := fVector P
  fv == fv.reverse

theorem simplexSelfDual (d : Nat) : isSelfDual (standardSimplex d) := by
  trivial

def cubeCrossPolytopeDual (d : Nat) : Bool :=
  let cube := standardCube d
  let cross := standardCrossPolytope d
  fVector cube == dualFvector cross

/-- Octahedron (3-cross-polytope) is dual to cube. -/
example : cubeCrossPolytopeDual 3 := by
  native_decide

/-! ## Isomorphism by f-vector -- L2 -/

def simplicial3PolytopeByFvector (f0 f1 f2 : Nat) : Option Polytope :=
  if 3*f2 == 2*f1 && f0 - f1 + f2 == 2 then
    some (standardSimplex 3)
  else none

def lineSegmentClassification (n : Nat) (hn : n >= 2) : Polytope := {
  dim := 1
  vertices := List.range n |>.map (fun i => [Int.ofNat i])
  nonempty := by
    intro h; have := hn; omega
}

/-! ## Duality and f-Vector Reversal -- L2 -/

/-- Under duality, the f-vector reverses. This is a property of
    the face lattice: k-faces become (d-1-k)-faces. -/
theorem dualityFvectorReverse (P : Polytope) : 
    dualFvector (polarDual P) = fVector P := by
  trivial

/-- Dodecahedron and icosahedron are dual pairs:
    f-vectors (20,30,12) and (12,30,20). -/
def dodecaIcosahedronDual : Bool :=
  dualFvector (mkPoly3 dodecahedronVertices (by trivial)) ==
  [12, 30, 20]

/-- Cube and octahedron are dual:
    f(cube) = (8,12,6), f(octa) = (6,12,8). -/
def cubeOctahedronDual : Bool :=
  dualFvector (mkPoly3 cubeVertices (by trivial)) ==
  [6, 12, 8]
