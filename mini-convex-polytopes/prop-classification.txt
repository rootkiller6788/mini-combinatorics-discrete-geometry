/-! ## f-Vector Validation for 3-Polytopes -- L4 -/

def isValid3DFvector (f0 f1 f2 : Nat) : Bool :=
  f0 - f1 + f2 == 2 &&
  3*f0 >= f1 + 6 &&
  3*f2 >= f1 + 6 &&
  f0 <= 2*f2 - 4 &&
  f2 <= 2*f0 - 4

#eval s!"(5,9,6) valid: {isValid3DFvector 5 9 6}"
#eval s!"(6,12,8) valid: {isValid3DFvector 6 12 8}"

/-! ## Regular Polytopes in Higher Dimensions -- L4 -/

def regularPolytopeCount (d : Nat) : Nat :=
  match d with
  | 0 => 1
  | 1 => 1
  | 2 => 0
  | 3 => 5
  | 4 => 6
  | _ => 3

def regular4Polytopes : List String := [

  "4-cross-polytope (16-cell) {3,3,4}: f=(8,24,32,16)",
  "24-cell {3,4,3}: f=(24,96,96,24) -- self-dual",
  "120-cell {5,3,3}: f=(600,1200,720,120)",
  "600-cell {3,3,5}: f=(120,720,1200,600)"
]

/-- Coxeter groups corresponding to regular polytopes.
    A_n: simplex, B_n: cube/cross-polytope, F_4: 24-cell,
    H_3: icosahedron/dodecahedron, H_4: 120/600-cell. -/
def coxeterGroupOfRegularPolytope (name : String) : String :=
  match name with
  | "simplex" => "A_n"
  | "cube" => "B_n"
  | "cross-polytope" => "B_n"
  | "icosahedron" => "H_3"
  | "dodecahedron" => "H_3"
  | "24-cell" => "F_4"
  | "120-cell" => "H_4"
  | "600-cell" => "H_4"
  | _ => "unknown"
