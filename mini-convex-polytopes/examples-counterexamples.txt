/-! ## Barnette Sphere: Non-polytopal 3-sphere -- L6 -/

/-- The Barnette sphere (1978) is a simplicial 3-sphere with
    8 vertices that is NOT combinatorially equivalent to the
    boundary of any 4-polytope. f-vector: (8, 27, 38, 19).
    This shows that not all simplicial spheres are polytopal.
    In contrast, all simplicial 2-spheres are polytopal (Steinitz). -/
#eval "── Barnette sphere: simplicial 3-sphere, non-polytopal ──"
#eval "f = (8, 27, 38, 19)"
#eval "Shows: simplicial sphere ≠ polytope boundary for d >= 4"

/-! ## Non-realizable f-Vectors -- L6 -/

/-- f-vector (6, 14, 16, 8) is NOT realizable as any
    4-polytope, despite satisfying Dehn-Sommerville equations
    and the Lower Bound Theorem. -/
def nonRealizableFvector : List Nat := [6, 14, 16, 8]

/-- f-vector (12, 36, 36, 12) satisfies necessary conditions
    but is NOT realizable. -/
def anotherNonRealizableFvector : List Nat := [12, 36, 36, 12]

#eval s!"Non-realizable 4D f-vectors:"
#eval s!"  {nonRealizableFvector}: satisfies DS + LBT but not realizable"
#eval s!"  {anotherNonRealizableFvector}: same issue"

/-! ## Hirsch Counterexample -- L6 -/

/-- Santos (2010) disproved the Hirsch conjecture by constructing
    a 43-dimensional polytope with 86 facets whose graph diameter
    is at least 44 > 86-43 = 43. -/
#eval "── Hirsch counterexample (Santos 2010) ──"
#eval "d=43, n=86 facets, diameter >= 44 > n-d = 43"
#eval "Showed that diameter can exceed n-d"

/-- A smaller counterexample (Matschke-Santos-Weibel 2015):
    A 20-dimensional polytope with 40 facets and diameter 21. -/
#eval "Smaller counterexample: d=20, n=40, diam >= 21 > 20"
