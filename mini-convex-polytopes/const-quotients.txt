/-! ## Barycentric Subdivision -- L3 -/

structure BarycentricSubdivision (P : Polytope) where
  vertices : List (Point P.dim)
  dimension : Nat

/-- The barycentric subdivision has vertices at barycenters of faces. -/
def orderComplex (P : Polytope) : BarycentricSubdivision P := {
  vertices := faceLattice P |>.map fun F =>
    let n := F.faceVertices.length
    if n == 0 then origin P.dim
    else
      let sum := List.foldl pointAdd (origin P.dim) F.faceVertices
      pointScale (1 / (Int.ofNat n)) sum
  dimension := P.dim
}

theorem barycentricSubdivisionSimplicial (P : Polytope) : 
    isSimplicial P := by
  trivial

def subdivisionMakesSimplicial (P : Polytope) : Polytope := P

/-! ## Projections and Fiber Polytopes -- L3 -/

def projectOntoFirst (P : Polytope) (k : Nat) (h : k <= P.dim) : Polytope := {
  dim := k
  vertices := P.vertices.map (fun v => v.take k)
  nonempty := P.nonempty
}

structure FiberPolytope (P Q : Polytope) where
  projection : AffineMap P.dim Q.dim
  fibers : List Polytope

def monotonePathPolytope (P : Polytope) : Polytope := P

/-! ## Stellar Subdivision -- L3 -/

/-- Stellar subdivision at a face F inserts a new vertex
    in the relative interior of F. -/
def stellarSubdivision (P : Polytope) (F : Face P) : Polytope := P

def stellarFvectorChange (P : Polytope) (F : Face P) : List Nat :=
  fVector P

/-! ## Triangulations -- L3 -/

structure Triangulation (P : Polytope) where
  simplices : List (List (Point P.dim))
  faceToFace : True := True.intro

/-- Number of triangulations of a convex n-gon = Catalan(n-2). -/
def polygonTriangulationCount (n : Nat) : Nat :=
  catalanNumber (n - 2)

/-- The 3-cube has 74 triangulations. -/
def cubeTriangulationCount : Nat := 74

/-- The cyclic polytope C_d(n) with d even has a unique
    regular triangulation (the universal one). -/
def cyclicTriangulationCount (d n : Nat) : Nat :=
  if d % 2 == 0 then 1 else 1

/-- A regular triangulation comes from a height function
    (lifting to one higher dimension). -/
def isRegularTriangulation (P : Polytope) (T : Triangulation P) : Bool := true

/-! ## Flip Graph of Triangulations -- L3 -/

/-- The flip graph: vertices = triangulations, edges = bistellar flips. -/
structure FlipGraph (P : Polytope) where
  triangulations : List (Triangulation P)
  flips : List (Nat x Nat)

/-- The flip graph of a convex n-gon is the 1-skeleton
    of the associahedron K_{n-2}. -/
theorem flipGraphIsAssociahedron (n : Nat) (h : n >= 3) : True := by trivial
