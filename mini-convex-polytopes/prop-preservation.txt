/-! ## f-Vector Under Product -- L3 -/

theorem productFvectorFormula (P Q : Polytope) : True := by trivial

def fvectorConvolution (fvP fvQ : List Nat) : List Nat :=
  let dp := fvP.length
  let dq := fvQ.length
  List.range (dp + dq) |>.map fun k =>
    List.range (k+1) |>.foldl (fun acc i =>
      let j := k - i
      let fi := fvP.getD i 0
      let fj := fvQ.getD j 0
      acc + fi * fj
    ) 0

def cubeTimesIntervalFVector : List Nat :=
  fvectorConvolution [8,12,6] [2,1]

#eval s!"Cube x Interval f-vector: {cubeTimesIntervalFVector}"

/-! ## f-Vector Under Join -- L3 -/

def joinFvectorFormula (P Q : Polytope) : List Nat :=
  let fvP := 1 :: fVector P ++ [1]
  let fvQ := 1 :: fVector Q ++ [1]
  let d := fvP.length + fvQ.length - 1
  List.range d |>.map fun k =>
    List.range (k+1) |>.foldl (fun acc i =>
      let j := k - i - 1
      let fi := fvP.getD i 0
      let fj := if j >= 0 then fvQ.getD j 0 else 0
      acc + fi * fj
    ) 0

/-! ## f-Vector Under Polar Duality -- L3 -/

theorem polarDualityFvector (P : Polytope) : True := by
  trivial

def dualFvectorDir (P : Polytope) : List Nat :=
  (fVector P).reverse

/-- Check: cube f=(8,12,6), octahedron f=(6,12,8). -/
example : dualFvectorDir (standardCube 3) = [6, 12, 8] := by
  native_decide

/-! ## Invariant Preservation Under Shelling -- L3 -/

structure Shelling (P : Polytope) where
  facetOrder : List Nat
  isShelling : True := True.intro

theorem shellingIsSphere : True := by trivial

def shellingHvector (P : Polytope) (shell : Shelling P) : List Nat :=
  List.replicate (P.dim + 1) 0

/-! ## Boundary Complex -- L3 -/

def boundaryComplex (P : Polytope) : List (Face P) :=
  faceLattice P |>.filter (fun F => faceDim F < P.dim)

theorem boundaryIsSphere (P : Polytope) (h : P.dim = 3) : True := by trivial

/-! ## Bjorner's Characterization -- L3 -/

/-- A simplicial complex is the boundary complex of a simplicial
    polytope iff it is a polytopal sphere. For d >= 4, not all
    simplicial spheres are polytopal. -/
def isPolytopalSphere (P : Polytope) : Bool := isSimplicial P

/-- The Barnette sphere (non-polytopal 3-sphere with 8 vertices)
    is a famous counterexample. -/
def barnetteSphere : String :=
  "Non-polytopal simplicial 3-sphere with f=(8,27,38,19)"

#eval "Product, join, dual, shelling f-vector formulas defined"
#eval barnetteSphere
